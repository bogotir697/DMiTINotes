$$\begin{align}
\exists \ \{(x_0,y_0)&, (x_1,y_1), \ldots, (x_n, y_n)\} \\
P_k(x) &= \frac{1}{x-x_k} \displaystyle \prod_{i=0}^n{(x-x_i)} \\
a_k\cdot P_k(x_k) &= y_k \Leftrightarrow a_k = \frac{y_k}{P_k(x_k)} \\
P(x) &= \displaystyle \sum_{k=0}^n{a_k\cdot P_k(x)}
\end{align}$$
