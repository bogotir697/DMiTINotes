#### [[Вопросы к Экзамену]]
---
$$\begin{align}
	&a=a_0b+r_0 \\
	&\quad\vdots\\
	&r_k=a_{k+1}r_{k+1}+r_{k+2}
\end{align}$$ $$\begin{array}{c|c}
	a&&1&0\\
	b&&0&1\\
	\hline
	r_0&a_0&1&a_0\\
	r_1&a_1&a_1&a_1a_0+1\\
	\vdots\\
	r_k&a_k&q_k&p_k\\
	\vdots\\
	r_{n-1}&a_{n-1}&q_{n-1}&p_{n-1}\\
	r_n=0&a_n&q_n&p_n
\end{array}\begin{align}
	p_k=a_kp_{k-1}+p_{k-2}\\
	q_k=a_kq_{k-1}+q_{k-2}
\end{align}$$
*Пример.* $$\begin{array}{c|c}
	9&&1&0\\
	7&&0&1\\
	\hline
	2&1&1&1\\
	1&3&3&4\\
	0&2&7&9
\end{array}$$
*Утв.* $a\cdot q_k - b\cdot p_k = (-1)^k\cdot r_k$
*Доказательство.* $$\begin{align}
	&\text{База: } k=-2: a\cdot1-b\cdot0=(-1)^{-2}\cdot a=a \\
	&\quad\quad\:\, k=-1: a\cdot0-b\cdot1=(-1)^{-1}\cdot b=-b \\
	&\text{Переход: } aq_k-b p_k = a\cdot a_k q_{k-1}+a\cdot q_{k-2} - b\cdot a_kp_{k-1}-b\cdot p_{k-2} = \\
	&\quad\quad\quad\quad=a_k(aq_{k-1}-bp_{k-1})+(aq_{k-2}-bp_{k-2})=\\
	&\quad\quad\quad\quad=a_k\cdot(-1)^{k-1}r_{k-1}+(-1)^{k-2}r_{k-2}=(-1)^kr_k\\
	&\quad\quad\quad\left[\begin{array}{l}
		r_{k-2}=a_k r_{k-1}+r_k \cdot (-1)^k\\
		(-1)^k r_{k-2}=a_k \cdot (-1)^k r_{k-1}+(-1)^k r_k\\
		(-1)^k r_{k-2}-a_k \cdot (-1)^k r_{k-1}=(-1)^k r_k\\
		(-1)^{k-2} r_{k-2}+a_k \cdot (-1)^{k-1} r_{k-1}=(-1)^k r_k
		\end{array}
		\right.
\end{align}$$
